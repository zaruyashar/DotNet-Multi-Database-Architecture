<script>
    // Auto-generate tracking number on page load
    function generateTrackingNumber() {
        const prefix = 'TRK-';
        const random = Math.random().toString(36).substring(2, 10).toUpperCase();
        return prefix + random;
    }

    // Set default date to now
    function setDefaultDate() {
        const now = new Date();
        const offset = now.getTimezoneOffset();
        const adjustedDate = new Date(now.getTime() - offset * 60 * 1000);
        const dateString = adjustedDate.toISOString().slice(0, 16);
        document.getElementById('dateInput').value = dateString;
    }

    // Initialize on page load
    window.addEventListener('load', function() {
        const trackingInput = document.getElementById('trackingInput');
        if (!trackingInput.value) {
            trackingInput.value = generateTrackingNumber();
            document.getElementById('previewTracking').textContent = trackingInput.value;
        }
        setDefaultDate();
    });

    // Preview updates
    const trackingInput = document.getElementById('trackingInput');
    const senderInput = document.getElementById('senderInput');
    const receiverInput = document.getElementById('receiverInput');
    const originInput = document.getElementById('originInput');
    const destinationInput = document.getElementById('destinationInput');
    const dateInput = document.getElementById('dateInput');
    const statusInput = document.getElementById('statusInput');

    trackingInput?.addEventListener('input', (e) => {
        document.getElementById('previewTracking').textContent = e.target.value || 'TRK-XXXXXXXX';
    });

    senderInput?.addEventListener('input', (e) => {
        document.getElementById('previewSender').textContent = e.target.value || '-';
    });

    receiverInput?.addEventListener('input', (e) => {
        document.getElementById('previewReceiver').textContent = e.target.value || '-';
    });

    originInput?.addEventListener('change', (e) => {
        document.getElementById('previewOrigin').textContent = e.target.value || 'Çıkış';
    });

    destinationInput?.addEventListener('change', (e) => {
        document.getElementById('previewDestination').textContent = e.target.value || 'Varış';
    });

    dateInput?.addEventListener('change', (e) => {
        if (e.target.value) {
            const date = new Date(e.target.value);
            const formatted = date.toLocaleDateString('tr-TR', {
                day: '2-digit',
                month: '2-digit',
                year: 'numeric',
                hour: '2-digit',
                minute: '2-digit'
            });
            document.getElementById('previewDate').textContent = formatted;
        }
    });

    statusInput?.addEventListener('change', (e) => {
        document.getElementById('previewStatus').textContent = e.target.value || 'Bekliyor';
    });

    // Form validation
    document.getElementById('shipmentForm')?.addEventListener('submit', (e) => {
        const tracking = trackingInput.value.trim();
        const sender = senderInput.value.trim();
        const receiver = receiverInput.value.trim();
        const origin = originInput.value;
        const destination = destinationInput.value;
        const status = statusInput.value;

        if (!tracking || !sender || !receiver || !origin || !destination || !status) {
            e.preventDefault();
            alert('Lütfen tüm zorunlu alanları doldurun!');
            return false;
        }

        if (origin === destination) {
            e.preventDefault();
            alert('Çıkış ve varış şehri aynı olamaz!');
            return false;
        }
    });
</script>
